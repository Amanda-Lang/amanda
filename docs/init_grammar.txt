program
  : block
  ;

block
  : ((expression|statement) SEMI )*
  ;

type
  : IDENTIFIER
  ;


simple_declaration
  : DECL type IDENTIFIER
  ;

array_declaration
  : VECTOR type IDENTIFIER LBRACKETS [ INTEGER ]
  ;

declaration
  : simple_declaration
  | array_declaration
  ;

statement
  :
  | declaration
  | function_declaration
  | mostra_statement
  ;

mostra_statement
  : MOSTRA expression
  ;

expression
  : term (add_operator term)*
  ;

term
  : factor (multi_operator term)*
  ;


actual_parameters
  : IDENTIFIER (COMMA IDENTIFIER)*
  ;

factor
  : INTEGER
  | LPAR expression RPAR
  | add_operator factor
  | IDENTIFIER (LPAR actual_parameters? RPAR)?
  ;

mult_operator
  : STAR
  | SLASH
  | MODULO
  ;

add_operator
  : PLUS
  | MINUS
  ;


formal_parameter_section
  : type IDENTIFIER (COMMA type IDENTIFIER)*
  ;


function_declaration
  : DEFINA IDENTIFIER LPAR formal_parameter_section? RPAR (DOUBLECOL type)? LBRACES function_block RBRACES
  ;
